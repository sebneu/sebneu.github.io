CKAN Revision Feed

CKAN provides an Atom feed to get updates on edited and created datasets and resources. By default, the feed can be found at {portal-url}/revision/list?format=atom.

In the Python code example I accessed the feed using the feedparser package. The titles of the feed entries are of the following form:

r863bfa65-bfed-40b1-b6ee-e78275be3283 [ra2014:updated:resources]: REST API: Update Objekt ra2014

I used the information within the square brackets: ra2014 is the internal dataset name, updated tells you if the dataset is created or updated and the third value, resources, appears only if a resource of the dataset has been edited.

import datetime
import feedparser
FEED = 'https://www.data.gv.at/katalog/revision/list?format=atom&#39;
def parse_feed(feed_url, days=30):
start_date = datetime.datetime.now() - datetime.timedelta(days=days)
d = feedparser.parse(feed_url)
total_ids = set()
created_ids = set()
updated_ids = set()
for post in d.entries:
date = datetime.datetime(*(post.updated_parsed[0:6]))
if date >= start_date:
t = post.title
s = find_between(t, '[', ']')
details = s.split(':')
if len(details) > 0:
id = details[0]
total_ids.add(id)
if len(details) > 1:
typ = details[1]
if typ == 'created':
created_ids.add(id)
elif typ == 'updated':
updated_ids.add(id)
print 'total number of edited datasets:', len(total_ids)
print 'number of created datasets:', len(created_ids)
print 'number of updated datasets:', len(updated_ids)
def find_between(s, first, last):
try:
start = s.index(first) + len(first)
end = s.index(last, start)
return s[start:end]
except ValueError:
return ""
if __name__ == '__main__':
parse_feed(FEED)
